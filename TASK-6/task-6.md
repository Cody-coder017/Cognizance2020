Intialize no

temp=no

sum=0

order= length of no

for( ;no>0;)

    digit=n%10
    sum=sum+digit**order
    no=no/10
END

IF sum==temp

    PRINT "IT IS AN ARMSTRONG NUMBER"
ELSE

    PRINT "IT IS NOT AN ARMSTRONG NUMBER"
